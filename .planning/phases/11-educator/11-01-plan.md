```typescript
import { useState, useEffect } from 'react';
import { supabase } from '../lib/supabase';
import type { EducatorProfile } from '../types/profile';

interface UseEducatorProfileReturn {
  profile: EducatorProfile | null;
  loading: boolean;
  error: string | null;
  updateProfile: (updates: Partial<Omit<EducatorProfile, 'id' | 'created_at' | 'updated_at' | 'verified'>>) => Promise<void>;
  refetch: () => Promise<void>;
}

export function useEducatorProfile(userId: string): UseEducatorProfileReturn {
  const [profile, setProfile] = useState<EducatorProfile | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchProfile = async () => {
    if (!userId) return;
    setLoading(true);
    setError(null);
    const { data, error: err } = await supabase
      .from('educator_profiles')
      .select('*')
      .eq('id', userId)
      .single();
    if (err && err.code !== 'PGRST116') {
      setError(err.message);
    } else {
      setProfile(data ?? null);
    }
    setLoading(false);
  };

  useEffect(() => { fetchProfile(); }, [userId]);

  const updateProfile = async (updates: Partial<Omit<EducatorProfile, 'id' | 'created_at' | 'updated_at' | 'verified'>>) => {
    setError(null);
    const { error: err } = await supabase
      .from('educator_profiles')
      .upsert({ id: userId, ...updates, updated_at: new Date().toISOString() });
    if (err) { setError(err.message); throw err; }
    await fetchProfile();
  };

  return { profile, loading, error, updateProfile, refetch: fetchProfile };
}
```

```typescript
import { useState, useEffect, useCallback } from 'react';
import { supabase } from '../lib/supabase'; // match existing import path
import type { Database } from '../types/database';

type ClassRow = Database['public']['Tables']['classes']['Row'];

interface UseClassManagementReturn {
  classes: ClassRow[];
  loading: boolean;
  error: string | null;
  createClass: (name: string, description?: string) => Promise<ClassRow>;
  archiveClass: (classId: string) => Promise<void>;
  enrollStudent: (classId: string, studentId: string) => Promise<void>;
  unenrollStudent: (classId: string, studentId: string) => Promise<void>;
  refetch: () => Promise<void>;
}

export function useClassManagement(educatorId: string): UseClassManagementReturn {
  const [classes, setClasses] = useState<ClassRow[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchClasses = useCallback(async () => {
    if (!educatorId) return;
    setLoading(true);
    setError(null);
    const { data, error: err } = await supabase
      .from('classes')
      .select('*')
      .eq('educator_id', educatorId)
      .order('created_at', { ascending: false });
    if (err) setError(err.message);
    else setClasses(data ?? []);
    setLoading(false);
  }, [educatorId]);

  useEffect(() => { fetchClasses(); }, [fetchClasses]);

  const createClass = async (name: string, description = '') => {
    const { data, error: err } = await supabase
      .from('classes')
      .insert({ name, description, educator_id: educatorId })
      .select()
      .single();
    if (err) { setError(err.message); throw err; }
    await fetchClasses();
    return data;
  };

  const archiveClass = async (classId: string) => {
    const { error: err } = await supabase
      .from('classes')
      .update({ archived: true })
      .eq('id', classId)
      .eq('educator_id', educatorId);
    if (err) { setError(err.message); throw err; }
    await fetchClasses();
  };

  const enrollStudent = async (classId: string, studentId: string) => {
    const { error: err } = await supabase
      .from('class_enrollments')
      .insert({ class_id: classId, student_id: studentId });
    if (err) { setError(err.message); throw err; }
  };

  const unenrollStudent = async (classId: string, studentId: string) => {
    const { error: err } = await supabase
      .from('class_enrollments')
      .delete()
      .eq('class_id', classId)
      .eq('student_id', studentId);
    if (err) { setError(err.message); throw err; }
  };

  return { classes, loading, error, createClass, archiveClass, enrollStudent, unenrollStudent, refetch: fetchClasses };
}
```

If `classes` table does not have an `archived` column, use a `status` column set to 'archived' instead. Check the existing migration for actual column names.
