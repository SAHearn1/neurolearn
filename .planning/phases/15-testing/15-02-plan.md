2. Create src/test/accessibility.test.tsx:

```typescript
import { render } from '@testing-library/react';
import { axe } from '../lib/axe-setup';
import { MemoryRouter } from 'react-router-dom';

// Import all page components
import HomePage from '../pages/HomePage';
import LoginPage from '../pages/LoginPage';
import SignUpPage from '../pages/SignUpPage';

// Mock Supabase to prevent real network calls
jest.mock('../lib/supabase', () => ({
  supabase: {
    auth: { onAuthStateChange: jest.fn(() => ({ data: { subscription: { unsubscribe: jest.fn() } } })) },
    from: jest.fn(() => ({ select: jest.fn(() => ({ data: [], error: null })) })),
  },
}));

const wrap = (ui: React.ReactNode) =>
  render(<MemoryRouter>{ui}</MemoryRouter>);

describe('Accessibility — Page components', () => {
  test('HomePage has no axe violations', async () => {
    const { container } = wrap(<HomePage />);
    const results = await axe(container);
    expect(results).toHaveNoViolations();
  });

  test('LoginPage has no axe violations', async () => {
    const { container } = wrap(<LoginPage />);
    const results = await axe(container);
    expect(results).toHaveNoViolations();
  });

  test('SignUpPage has no axe violations', async () => {
    const { container } = wrap(<SignUpPage />);
    const results = await axe(container);
    expect(results).toHaveNoViolations();
  });
});
````

Check package.json — if jest-axe is not installed, add it to devDependencies in package.json and run npm install. Also check that vitest is configured to work with jest-axe (may need expect.extend(toHaveNoViolations())).

If using Vitest (not Jest), use @axe-core/react or configure jest-axe for Vitest:

- Add `import { toHaveNoViolations } from 'jest-axe'; expect.extend(toHaveNoViolations);` in test setup

```yaml
  runs-on: ubuntu-latest
  needs: [test] # Replace 'test' with the actual job name from existing ci.yml
  steps:
    - uses: actions/checkout@v4
    - uses: actions/setup-node@v4
      with:
        node-version: '20'
        cache: 'npm'
    - run: npm ci
    - run: npm run build
    - name: Run Lighthouse CI
      run: |
        npm install -g @lhci/cli@0.14.x
        lhci autorun
      env:
        LHCI_GITHUB_APP_TOKEN: ${{ secrets.LHCI_GITHUB_APP_TOKEN }}
        VITE_SUPABASE_URL: ${{ secrets.VITE_SUPABASE_URL }}
        VITE_SUPABASE_ANON_KEY: ${{ secrets.VITE_SUPABASE_ANON_KEY }}
```
