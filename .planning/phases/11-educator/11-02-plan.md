```tsx
import { useAuthStore } from '../store/authStore';
import { useEducatorProfile } from '../hooks/useEducatorProfile';
import { useClassManagement } from '../hooks/useClassManagement';
import { ClassList } from '../components/educator/ClassList';
import { Spinner } from '../components/ui/Spinner';
import { Card } from '../components/ui/Card';

export default function EducatorDashboardPage() {
  const { user } = useAuthStore();
  const { profile, loading: profileLoading } = useEducatorProfile(user?.id ?? '');
  const { classes, loading: classesLoading, createClass, archiveClass } = useClassManagement(user?.id ?? '');

  if (profileLoading || classesLoading) return <Spinner />;

  const totalStudents = classes.reduce((sum, c) => sum + (c.student_count ?? 0), 0);

  return (
    <main>
      <h1>Educator Dashboard{profile ? ` — ${profile.school_name ?? ''}` : ''}</h1>
      <div role="region" aria-label="Quick stats">
        <Card><span>{classes.length} Classes</span></Card>
        <Card><span>{totalStudents} Students</span></Card>
        <Card><span>{classes.filter(c => !c.archived).length} Active</span></Card>
      </div>
      {classes.length === 0
        ? <p>No classes yet. <button onClick={() => createClass('New Class')}>Create your first class</button></p>
        : <ClassList classes={classes} onArchive={archiveClass} onCreate={createClass} />
      }
    </main>
  );
}
```
