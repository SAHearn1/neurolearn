import * as Sentry from '@sentry/react';

// Initialize Sentry — only when DSN is configured
if (import.meta.env.VITE_SENTRY_DSN) {
  Sentry.init({
    dsn: import.meta.env.VITE_SENTRY_DSN,
    environment: import.meta.env.VITE_SENTRY_ENVIRONMENT ?? 'development',
    // Only enable in production-like environments
    enabled: import.meta.env.VITE_SENTRY_ENVIRONMENT === 'production',
    integrations: [
      Sentry.browserTracingIntegration(),
      Sentry.replayIntegration({
        maskAllText: true, // Privacy: mask PII in replays
        blockAllMedia: false,
      }),
    ],
    tracesSampleRate: import.meta.env.VITE_SENTRY_ENVIRONMENT === 'production' ? 0.1 : 0,
    replaysSessionSampleRate: 0.1,
    replaysOnErrorSampleRate: 1.0,
  });
}
```

// Vercel Analytics custom events
// Uses window.va if available (injected by Vercel), no-ops otherwise
function track(event: string, properties?: Record<string, string | number | boolean>): void {
  if (typeof window !== 'undefined' && (window as Record<string, unknown>).va) {
    (window as Record<string, unknown & { track: typeof track }>).va?.('event', { name: event, ...properties });
  }
}

export const analytics = {
  lessonCompleted: (lessonId: string, timeSpentSeconds: number) =>
    track('lesson_completed', { lessonId, timeSpentSeconds }),

  racaSessionEnded: (sessionId: string, finalState: string, durationSeconds: number) =>
    track('raca_session_ended', { sessionId, finalState, durationSeconds }),

  pageViewed: (page: string) =>
    track('page_viewed', { page }),
};
```
