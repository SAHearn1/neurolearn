```typescript
.select('\*')
.eq('id', userId)
.single();

else setProfile(data ?? null);
setLoading(false);
};

useEffect(() => { fetchProfile(); }, [userId]);

const updateProfile = async (updates: Partial<Omit<ParentProfile, 'id' | 'created_at' | 'updated_at'>>) => {
const { error: err } = await supabase
.from('parent_profiles')
.upsert({ id: userId, ...updates, updated_at: new Date().toISOString() });
if (err) throw err;
await fetchProfile();
};

return { profile, loading, error, updateProfile, refetch: fetchProfile };
}
````

```typescript
import { useState, useEffect, useCallback } from 'react';
import { supabase } from '../lib/supabase';
import type { ParentStudentLink } from '../types/profile';

export function useParentStudentLinks(parentId: string) {
  const [links, setLinks] = useState<ParentStudentLink[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchLinks = useCallback(async () => {
    if (!parentId) return;
    setLoading(true);
    const { data, error: err } = await supabase
      .from('parent_student_links')
      .select('*, profiles!parent_student_links_student_id_fkey(full_name, avatar_url)')
      .eq('parent_id', parentId)
      .order('created_at', { ascending: false });
    if (err) setError(err.message);
    else setLinks((data ?? []) as ParentStudentLink[]);
    setLoading(false);
  }, [parentId]);

  useEffect(() => { fetchLinks(); }, [fetchLinks]);

  const requestLink = async (studentEmail: string) => {
    // Look up student by email in profiles
    const { data: student, error: lookupErr } = await supabase
      .from('profiles')
      .select('id')
      .eq('email', studentEmail)
      .single();
    if (lookupErr || !student) throw new Error('Student not found');
    const { error: err } = await supabase
      .from('parent_student_links')
      .insert({ parent_id: parentId, student_id: student.id, status: 'pending' });
    if (err) throw err;
    await fetchLinks();
  };

  const approveLink = async (linkId: string) => {
    const { error: err } = await supabase
      .from('parent_student_links')
      .update({ status: 'approved' })
      .eq('id', linkId);
    if (err) throw err;
    await fetchLinks();
  };

  const removeLink = async (linkId: string) => {
    const { error: err } = await supabase
      .from('parent_student_links')
      .delete()
      .eq('id', linkId)
      .eq('parent_id', parentId);
    if (err) throw err;
    await fetchLinks();
  };

  return { links, loading, error, requestLink, approveLink, removeLink, refetch: fetchLinks };
}
````
